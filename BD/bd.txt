alter procedure vehiculoeventos
(
@ind int ,
@IdVehiculo int ,
@descripcion varchar(100),
@idmarcar int, 
@idmodelo int ,
@idcolor int,
@precio decimal(10,4),
@anio int,
@Motor varchar(100),
@Ruedas varchar(20)
)
as

begin
 if @ind = 1
 begin
 insert into Vehiculo
 (
  
	Descripcion ,	IdMarca ,IdModelo ,
       IdColor ,	PrecioVenta ,
        Anio ,     Motor ,    Ruedas ,
	Estatus 
	)
	values 
	(
	@descripcion,@idmarcar , @idmodelo,
	@idcolor,@precio,
	@anio , @Motor, @Ruedas,
	1

	)

 end

  if @ind = 2
 begin
  update Vehiculo
  set 
  	Descripcion = @descripcion ,
	IdMarca  = @idmarcar,
	IdModelo = @idmodelo ,
       IdColor= @idcolor ,
	   PrecioVenta = @precio ,
        Anio = @anio ,  
		Motor = @Motor ,  
		Ruedas = @Ruedas   
  where  IdVehiculo = @IdVehiculo
 end
  if @ind = 3
 begin
  delete from Vehiculo
  where IdVehiculo = @IdVehiculo
 end

end
